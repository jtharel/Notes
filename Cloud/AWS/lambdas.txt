# Manually list Lambda functions and their roles at once:
aws lambda list-functions --query "Functions[*].{Name:FunctionName, Role:Role}" --output table

	# From the role names returned (role name is after the arn), inspect its permissions
	aws iam list-attached-role-policies --role-name "<ROLE-NAME>" --profile <PROFILE_NAME>

	# automate checking all permissions:
	for perms in `aws lambda list-functions --query "Functions[*].{Name:FunctionName, Role:Role}" --output json --profile <PROFILE_NAME> |jq -r '.[] | .Role | split("/") | last'` ; do
		echo "=== $perms ==="
		aws iam list-attached-role-policies --role-name $perms --profile <PROFILE_NAME>
		echo "=== $perms ===" 
		echo " "
	done

	# Also check Inline policies (may not work):
	aws iam list-role-policies --role-name "<role-name>"
