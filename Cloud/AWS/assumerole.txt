	#Script to show all AssumeRoles that do not have an ExternalID
	#!/bin/bash                                                                                                        
	for role in $(aws iam list-roles --query "Roles[].RoleName" --output text --profile <PROFILE_NAME> | tr '\t' '\n'); do                      
  		policy=$(aws iam get-role --role-name "$role" --query "Role.AssumeRolePolicyDocument" --output json --profile <PROFILE_NAME>)             
  		if echo "$policy" | grep -q "AssumeRole" && ! echo "$policy" | grep -q "<ACCOUNT NUMBER>" && ! echo "$policy" | grep -q "ExternalId"; then 
                	echo "CROSS-ACCOUNT, MISSING ExternalId: $role"                                                                
			echo "$policy"                                                                                                 
    			echo "---"                                        
  		fi                                                                                                               
	done   
	

	#Script to show all AssumeRoles that have the ExternalID set
	#!/bin/bash
	for role in $(aws iam list-roles --query "Roles[].RoleName" --output text --profile <PROFILE_NAME> | tr '\t' '\n'); do                      
		policy=$(aws iam get-role --role-name "$role" --query "Role.AssumeRolePolicyDocument" --output json --profile <PROFILE_NAME>)             
		if echo "$policy" | grep -q "ExternalId"; then                                                                   
			echo "HAS ExternalId: $role"                                                                                   
			echo "$policy"                                                                                                 
			echo "---"                                                                                                     
	  	fi
	done
